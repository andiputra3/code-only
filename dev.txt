"""ST-LMS v2.1 Health Monitoring Dashboard.

Real-time monitoring of:
- Circuit breaker states
- System health (DB, memory, disk)
- Pipeline success rates
- Extension module health
- Error rates and patterns
- Performance metrics
"""
import streamlit as st
import sqlite3
import psutil
import os
import json
import time
from datetime import datetime, timedelta
from typing import Dict, List, Optional
from dataclasses import dataclass

# Import dari v2.1 safety modules
from st_lms.safety.circuit_breaker import CIRCUIT_REGISTRY
from st_lms.safety.health_monitor import HealthMonitor, HealthStatus
from st_lms.monitoring.metrics_collector import MetricsCollector


@dataclass
class DashboardConfig:
    """Dashboard configuration."""
    db_path: str = "st_lms_v2.db"
    refresh_interval: int = 30  # seconds
    retention_days: int = 30


class HealthDashboard:
    """Main health monitoring dashboard."""
    
    def __init__(self, config: DashboardConfig = None):
        self.config = config or DashboardConfig()
        self.metrics = MetricsCollector(self.config.db_path)
        
    def render(self):
        """Render the complete dashboard."""
        st.set_page_config(
            page_title="ST-LMS v2.1 Health Monitor",
            page_icon="🏥",
            layout="wide",
            initial_sidebar_state="expanded"
        )
        
        # Header
        st.title("🏥 ST-LMS v2.1 Health Monitor")
        st.markdown("---")
        
        # Auto-refresh
        if st.checkbox("Auto-refresh", value=True):
            time.sleep(self.config.refresh_interval)
            st.experimental_rerun()
        
        # Main metrics row
        self._render_system_overview()
        
        st.markdown("---")
        
        # Tabs for detailed views
        tab1, tab2, tab3, tab4, tab5 = st.tabs([
            "🔌 Circuit Breakers",
            "💾 Database Health", 
            "📊 Pipeline Performance",
            "🧩 Extension Health",
            "⚠️ Error Log"
        ])
        
        with tab1:
            self._render_circuit_breakers()
        
        with tab2:
            self._render_database_health()
        
        with tab3:
            self._render_pipeline_performance()
        
        with tab4:
            self._render_extension_health()
        
        with tab5:
            self._render_error_log()
    
    def _render_system_overview(self):
        """Render top-level system metrics."""
        col1, col2, col3, col4, col5 = st.columns(5)
        
        with col1:
            # Overall health status
            health = self._get_overall_health()
            status_color = {
                HealthStatus.HEALTHY: "🟢",
                HealthStatus.DEGRADED: "🟡",
                HealthStatus.UNHEALTHY: "🟠",
                HealthStatus.CRITICAL: "🔴"
            }.get(health, "⚪")
            
            st.metric(
                "System Status",
                f"{status_color} {health.value}",
                delta=None
            )
        
        with col2:
            # Circuit breakers
            cb_stats = CIRCUIT_REGISTRY.get_all_stats()
            open_count = sum(1 for s in cb_stats.values() 
                           if s.state.value == "OPEN")
            total_count = len(cb_stats)
            
            st.metric(
                "Circuit Breakers",
                f"{total_count - open_count}/{total_count}",
                delta=f"{open_count} open" if open_count > 0 else None,
                delta_color="inverse" if open_count > 0 else "normal"
            )
        
        with col3:
            # Pipeline success rate (last hour)
            success_rate = self.metrics.get_pipeline_success_rate(hours=1)
            st.metric(
                "Pipeline Success",
                f"{success_rate:.1f}%",
                delta=self._get_success_rate_delta(hours=1)
            )
        
        with col4:
            # Memory usage
            mem = psutil.virtual_memory()
            st.metric(
                "Memory Usage",
                f"{mem.percent:.1f}%",
                delta=f"{mem.available / (1024**3):.1f} GB free"
            )
        
        with col5:
            # Active plans
            active_plans = self.metrics.get_active_plan_count()
            st.metric(

                delta=self.metrics.get_plan_change_delta()

        cb_stats = CIRCUIT_REGISTRY.get_all_stats()
        
        if not cb_stats:
            st.info("No circuit breakers registered yet.")
            return
        
        # Create table
        data = []
        for name, stats in cb_stats.items():
            data.append({
                "Name": name,
                "State": stats.state.value,
                "Failures": stats.failure_count,
                "Successes": stats.success_count,
                "Last Failure": self._format_timestamp(stats.last_failure_time),
                "Last Change": self._format_timestamp(stats.last_state_change),
            })
        
        # Color-code by state
        df = pd.DataFrame(data)
        
        def highlight_state(val):
            colors = {
                'CLOSED': 'background-color: #90EE90',
                'HALF_OPEN': 'background-color: #FFD700',
                'OPEN': 'background-color: #FF6B6B'
            }
            return colors.get(val, '')
        
        st.dataframe(
            df.style.applymap(highlight_state, subset=['State']),
            use_container_width=True

        # Summary
        col1, col2, col3 = st.columns(3)
        with col1:
            closed = sum(1 for s in cb_stats.values() if s.state.value == "CLOSED")
            st.metric("Closed (Healthy)", closed)
        with col2:
            half_open = sum(1 for s in cb_stats.values() if s.state.value == "HALF_OPEN")
            st.metric("Half-Open (Testing)", half_open)
        with col3:
            open_count = sum(1 for s in cb_stats.values() if s.state.value == "OPEN")

        st.subheader("💾 Database Health")
        
        col1, col2, col3 = st.columns(3)
        
        with col1:

                "FK Violations",
                fk_violations,
                delta=None,
                delta_color="inverse" if fk_violations > 0 else "normal"

        st.markdown("---")

        if table_sizes:

            st.dataframe(df, use_container_width=True)
        
        # Recent activity
        st.markdown("---")
        st.subheader("Recent Activity (Last 24h)")
        
        activity = self.metrics.get_recent_activity(hours=24)

            df = pd.DataFrame(activity)

        """Render pipeline performance metrics."""
        st.subheader("📊 Pipeline Performance")

                index=3
            )
        
        with col2:
            # Success rate trend
            success_rates = self.metrics.get_success_rate_trend(hours=hours)
            if success_rates:
                st.line_chart(success_rates)
        
        st.markdown("---")
        
        # Stage performance
        st.subheader("Stage Performance")
        stage_stats = self.metrics.get_stage_performance(hours=hours)
        
        if stage_stats:
            df = pd.DataFrame(stage_stats)
            st.dataframe(df, use_container_width=True)
            
            # Duration chart
            st.bar_chart(df.set_index('Stage')['Avg Duration (ms)'])
        
        # Error breakdown
        st.markdown("---")
        st.subheader("Error Breakdown")
        
        errors = self.metrics.get_error_breakdown(hours=hours)
        if errors:
            df = pd.DataFrame(errors)
            st.bar_chart(df.set_index('Stage')['Error Count'])
    
    def _render_extension_health(self):
        """Render extension module health."""
        st.subheader("🧩 Extension Module Health")
        
        extensions = [
            'exit_manager',
            'reentry_manager', 
            'position_manager',
            'risk_analyzer',
            'portfolio_manager',
            'pattern_executor',
            'performance_tracker',
            'regime_adapter'
        ]
        
        cols = st.columns(4)
        for i, ext_name in enumerate(extensions):
            with cols[i % 4]:
                health = self.metrics.get_extension_health(ext_name)
                status_icon = "✅" if health['status'] == 'HEALTHY' else "⚠️"
                
                st.markdown(f"""
                **{status_icon} {ext_name.replace('_', ' ').title()}**
                - Status: {health['status']}
                - Calls: {health['call_count']}
                - Errors: {health['error_count']}
                - Success Rate: {health['success_rate']:.1f}%
                """)
    
    def _render_error_log(self):
        """Render error log."""
        st.subheader("⚠️ Recent Errors")
        
        # Filters
        col1, col2 = st.columns(2)
        with col1:
            error_types = st.multiselect(
                "Error Types",
                ["DB_ERROR", "ENGINE_ERROR", "VALIDATION_ERROR", "TIMEOUT_ERROR"],
                default=[]
            )
        
        with col2:
            limit = st.selectbox("Limit", [10, 25, 50, 100], index=1)
        
        errors = self.metrics.get_recent_errors(
            limit=limit,
            error_types=error_types if error_types else None
        )
        
        if errors:
            for error in errors:
                with st.expander(f"❌ {error['error_type']} - {error['timestamp']}"):
                    st.markdown(f"**Source:** {error['source']}")
                    st.markdown(f"**Message:** {error['message']}")
                    if error.get('traceback'):
                        st.code(error['traceback'], language='python')
        else:
            st.success("No errors found! 🎉")
    
    def _get_overall_health(self) -> HealthStatus:
        """Get overall system health status."""
        # Check circuit breakers
        cb_stats = CIRCUIT_REGISTRY.get_all_stats()
        open_count = sum(1 for s in cb_stats.values() if s.state.value == "OPEN")
        
        if open_count > 3:
            return HealthStatus.CRITICAL
        elif open_count > 1:
            return HealthStatus.UNHEALTHY
        
        # Check pipeline success rate
        success_rate = self.metrics.get_pipeline_success_rate(hours=1)
        if success_rate < 80:
            return HealthStatus.UNHEALTHY
        elif success_rate < 95:
            return HealthStatus.DEGRADED
        
        # Check memory
        mem = psutil.virtual_memory()
        if mem.percent > 95:
            return HealthStatus.CRITICAL
        elif mem.percent > 85:
            return HealthStatus.UNHEALTHY
        elif mem.percent > 75:
            return HealthStatus.DEGRADED
        
        return HealthStatus.HEALTHY
    
    def _format_timestamp(self, ts: Optional[datetime]) -> str:
        """Format timestamp for display."""
        if not ts:
            return "Never"
        
        now = datetime.utcnow()
        diff = now - ts
        
        if diff.total_seconds() < 60:
            return "Just now"
        elif diff.total_seconds() < 3600:
            return f"{int(diff.total_seconds() / 60)}m ago"
        elif diff.total_seconds() < 86400:
            return f"{int(diff.total_seconds() / 3600)}h ago"
        else:
            return ts.strftime("%Y-%m-%d %H:%M")
    
    def _get_success_rate_delta(self, hours: int) -> Optional[str]:
        """Get success rate change delta."""
        current = self.metrics.get_pipeline_success_rate(hours=hours)
        previous = self.metrics.get_pipeline_success_rate(
            hours=hours, offset_hours=hours
        )
        
        if previous is None:
            return None
        
        delta = current - previous
        if delta > 0:
            return f"+{delta:.1f}%"
        elif delta < 0:
            return f"{delta:.1f}%"
        return None


def main():
    """Main entry point for health dashboard."""
    import argparse
    
    parser = argparse.ArgumentParser(description="ST-LMS v2.1 Health Dashboard")
    parser.add_argument('--db-path', default='st_lms_v2.db',
                       help='Path to database')
    parser.add_argument('--port', type=int, default=8504,
                       help='Port to run dashboard on')
    
    args = parser.parse_args()
    
    config = DashboardConfig(db_path=args.db_path)
    dashboard = HealthDashboard(config)
    dashboard.render()


if __name__ == '__main__':
    main()

"""Metrics Collector for Health Dashboard.

Aggregates metrics from various sources for dashboard display.
"""
import sqlite3
from datetime import datetime, timedelta
from typing import Dict, List, Optional, Tuple


class MetricsCollector:
    """Collect and aggregate system metrics."""
    
    def __init__(self, db_path: str):
        self.db_path = db_path
    
    def _get_conn(self) -> sqlite3.Connection:
        """Get read-only database connection."""
        conn = sqlite3.connect(f"file:{self.db_path}?mode=ro", uri=True)
        conn.row_factory = sqlite3.Row
        return conn
    
    def get_pipeline_success_rate(self, hours: int = 1, 
                                 offset_hours: int = 0) -> float:
        """Get pipeline success rate for time period."""
        conn = self._get_conn()
        
        start_time = (
            datetime.utcnow() - timedelta(hours=hours + offset_hours)
        ).isoformat()
        end_time = (
            datetime.utcnow() - timedelta(hours=offset_hours)
        ).isoformat()
        
        try:
            row = conn.execute("""
                SELECT 
                    COUNT(*) as total,
                    SUM(CASE WHEN telemetry_status = 'SUCCESS' THEN 1 ELSE 0 END) as success
                FROM pipeline_telemetry
                WHERE telemetry_timestamp BETWEEN ? AND ?
                AND telemetry_stage_order = 0
            """, (start_time, end_time)).fetchone()
            
            if not row or row['total'] == 0:
                return 0.0
            
            return (row['success'] / row['total']) * 100
        
        finally:
            conn.close()
    
    def get_active_plan_count(self) -> int:
        """Get count of active trading plans."""
        conn = self._get_conn()
        
        try:
            row = conn.execute("""
                SELECT COUNT(*) as count
                FROM adaptive_plans
                WHERE plan_state NOT IN ('FINISHED', 'LEARNING')
            """).fetchone()
            
            return row['count'] if row else 0
        
        finally:
            conn.close()
    
    def get_plan_change_delta(self) -> Optional[str]:
        """Get plan count change vs 24h ago."""
        current = self.get_active_plan_count()
        
        conn = self._get_conn()
        try:
            yesterday = (datetime.utcnow() - timedelta(hours=24)).isoformat()
            row = conn.execute("""
                SELECT COUNT(*) as count
                FROM adaptive_plans
                WHERE plan_state NOT IN ('FINISHED', 'LEARNING')
                AND plan_updated_at < ?
            """, (yesterday,)).fetchone()
            
            yesterday_count = row['count'] if row else 0
            delta = current - yesterday_count
            
            if delta > 0:
                return f"+{delta}"
            elif delta < 0:
                return f"{delta}"
            return None
        
        finally:
            conn.close()
    
    def get_table_count(self) -> int:
        """Get total table count."""
        conn = self._get_conn()
        
        try:
            row = conn.execute("""
                SELECT COUNT(*) as count
                FROM sqlite_master
                WHERE type='table' AND name NOT LIKE 'sqlite_%'
            """).fetchone()
            
            return row['count'] if row else 0
        
        finally:
            conn.close()
    
    def get_fk_violation_count(self) -> int:
        """Get foreign key violation count."""
        conn = self._get_conn()
        
        try:
            rows = conn.execute("PRAGMA foreign_key_check").fetchall()
            return len(rows)
        
        finally:
            conn.close()
    
    def get_table_sizes(self, limit: int = 10) -> List[Tuple[str, int, float]]:
        """Get table sizes (name, rows, size_kb)."""
        conn = self._get_conn()
        
        try:
            tables = conn.execute("""
                SELECT name FROM sqlite_master
                WHERE type='table' AND name NOT LIKE 'sqlite_%'
            """).fetchall()
            
            sizes = []
            for table in tables:
                name = table['name']
                
                # Get row count
                row = conn.execute(
                    f"SELECT COUNT(*) as count FROM {name}"
                ).fetchone()
                count = row['count'] if row else 0
                
                # Estimate size (rough approximation)
                size_kb = count * 0.5  # ~500 bytes per row average
                
                sizes.append((name, count, size_kb))
            
            # Sort by size descending
            sizes.sort(key=lambda x: x[2], reverse=True)
            
            return sizes[:limit]
        
        finally:
            conn.close()
    
    def get_recent_activity(self, hours: int = 24) -> List[Dict]:
        """Get recent table activity."""
        conn = self._get_conn()
        
        cutoff = (datetime.utcnow() - timedelta(hours=hours)).isoformat()
        
        activity_tables = [
            ('adaptive_plans', 'plan_updated_at'),
            ('execution_tracks', 'execution_entry_timestamp'),
            ('pattern_instances', 'instance_timestamp'),
            ('supertrend_lines', 'line_start_timestamp'),
        ]
        
        activity = []
        try:
            for table, timestamp_col in activity_tables:
                row = conn.execute(f"""
                    SELECT COUNT(*) as count
                    FROM {table}
                    WHERE {timestamp_col} > ?
                """, (cutoff,)).fetchone()
                
                activity.append({
                    'Table': table,
                    'Count': row['count'] if row else 0
                })
            
            return activity
        
        finally:
            conn.close()
    
    def get_success_rate_trend(self, hours: int = 24) -> List[Dict]:
        """Get success rate trend over time."""
        conn = self._get_conn()
        
        cutoff = (datetime.utcnow() - timedelta(hours=hours)).isoformat()
        
        try:
            rows = conn.execute("""
                SELECT 
                    DATE(telemetry_timestamp) as date,
                    COUNT(*) as total,
                    SUM(CASE WHEN telemetry_status = 'SUCCESS' THEN 1 ELSE 0 END) as success
                FROM pipeline_telemetry
                WHERE telemetry_timestamp > ?
                AND telemetry_stage_order = 0
                GROUP BY DATE(telemetry_timestamp)
                ORDER BY date
            """, (cutoff,)).fetchall()
            
            return [
                {
                    'Date': row['date'],
                    'Success Rate': (row['success'] / row['total'] * 100) 
                                   if row['total'] > 0 else 0
                }
                for row in rows
            ]
        
        finally:
            conn.close()
    
    def get_stage_performance(self, hours: int = 24) -> List[Dict]:
        """Get performance metrics per pipeline stage."""
        conn = self._get_conn()
        
        cutoff = (datetime.utcnow() - timedelta(hours=hours)).isoformat()
        
        try:
            rows = conn.execute("""
                SELECT 
                    telemetry_stage as Stage,
                    COUNT(*) as Count,
                    AVG(telemetry_duration_ms) as Avg_Duration,
                    SUM(CASE WHEN telemetry_status = 'ERROR' THEN 1 ELSE 0 END) as Errors
                FROM pipeline_telemetry
                WHERE telemetry_timestamp > ?
                GROUP BY telemetry_stage
                ORDER BY Avg_Duration DESC
            """, (cutoff,)).fetchall()
            
            return [
                {
                    'Stage': row['Stage'],
                    'Count': row['Count'],
                    'Avg Duration (ms)': round(row['Avg_Duration'], 2),
                    'Errors': row['Errors']
                }
                for row in rows
            ]
        
        finally:
            conn.close()
    
    def get_error_breakdown(self, hours: int = 24) -> List[Dict]:
        """Get error breakdown by stage."""
        conn = self._get_conn()
        
        cutoff = (datetime.utcnow() - timedelta(hours=hours)).isoformat()
        
        try:
            rows = conn.execute("""
                SELECT 
                    telemetry_stage as Stage,
                    COUNT(*) as Error_Count
                FROM pipeline_telemetry
                WHERE telemetry_timestamp > ?
                AND telemetry_status = 'ERROR'
                GROUP BY telemetry_stage
                ORDER BY Error_Count DESC
            """, (cutoff,)).fetchall()
            
            return [dict(row) for row in rows]
        
        finally:
            conn.close()
    
    def get_extension_health(self, extension_name: str) -> Dict:
        """Get health metrics for an extension module."""
        conn = self._get_conn()
        
        cutoff = (datetime.utcnow() - timedelta(hours=24)).isoformat()
        
        try:
            # Count calls from circuit breaker stats
            from st_lms.safety.circuit_breaker import CIRCUIT_REGISTRY
            
            cb_name = f"ext_{extension_name}"
            stats = CIRCUIT_REGISTRY.get_all_stats().get(cb_name)
            
            if not stats:
                return {
                    'status': 'UNKNOWN',
                    'call_count': 0,
                    'error_count': 0,
                    'success_rate': 0.0
                }
            
            total = stats.success_count + stats.failure_count
            success_rate = (stats.success_count / total * 100) if total > 0 else 0
            
            status = 'HEALTHY'
            if stats.state.value == 'OPEN':
                status = 'CRITICAL'
            elif stats.state.value == 'HALF_OPEN':
                status = 'DEGRADED'
            elif success_rate < 90:
                status = 'DEGRADED'
            
            return {
                'status': status,
                'call_count': total,
                'error_count': stats.failure_count,
                'success_rate': success_rate
            }
        
        finally:
            conn.close()
    
    def get_recent_errors(self, limit: int = 25,
                         error_types: List[str] = None) -> List[Dict]:
        """Get recent errors from error_log."""
        conn = self._get_conn()
        
        try:
            if error_types:
                placeholders = ','.join('?' * len(error_types))
                rows = conn.execute(f"""
                    SELECT * FROM error_log
                    WHERE error_type IN ({placeholders})
                    ORDER BY error_timestamp DESC
                    LIMIT ?
                """, error_types + [limit]).fetchall()
            else:
                rows = conn.execute("""
                    SELECT * FROM error_log
                    ORDER BY error_timestamp DESC
                    LIMIT ?
                """, (limit,)).fetchall()
            
            return [dict(row) for row in rows]
        
        finally:
            conn.close()

"""ST-LMS v2.1 REST API.

Provides RESTful interface for external integrations.
"""

from functools import wraps

    level=logging.INFO,

)

class STLMSAPI:

                 api_key: str = None):

    def require_api_key(self, f: Callable) -> Callable:
        """Decorator to require API key authentication."""
        @wraps(f)
        def decorated(*args, **kwargs):
            if not self.api_key:
                # No API key configured, allow all
                return f(*args, **kwargs)
            
            api_key = request.headers.get('X-API-Key')
            if not api_key or api_key != self.api_key:
                return jsonify({'error': 'Invalid or missing API key'}), 401
            
            return f(*args, **kwargs)
        
        return decorated
